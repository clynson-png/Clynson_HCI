section Section1;

shared TB_HCI = let
    FonteInput = Excel.CurrentWorkbook(){[Name="INPUT"]}[Content],

    FonteBase =
        Table.SelectRows(
            FonteInput,
            each
                Text.Upper(Text.From([STATUS_EVENTO])) <> "STATUS_EVENTO"
                and Text.Upper(Text.From([SERIE])) <> "SERIE"
        ),

    Tipos =
        Table.TransformColumnTypes(
            FonteBase,
            {
                {"DATA_COLETA", type text},
                {"PROVA", type text},
                {"ATLETA", type text},
                {"EVENTO", type text},
                {"SESSAO", type text},
                {"ID_BLOCO", type text},
                {"STATUS_EVENTO", type text},
                {"TEMPO_MIN", type number},
                {"SERIE", type text},
                {"T1", type number},
                {"T2", type number},
                {"T3", type number},
                {"T4", type number},
                {"T5", type number},
                {"T6", type number},
                {"T7", type number},
                {"T8", type number},
                {"T9", type number},
                {"T10", type number}
            },
            "pt-BR"
        ),

    // Função: lista dos tiros da série
    ListaTiros = (linha as record) as list =>
        {
            try Number.From(Record.Field(linha, "T1")) otherwise null,
            try Number.From(Record.Field(linha, "T2")) otherwise null,
            try Number.From(Record.Field(linha, "T3")) otherwise null,
            try Number.From(Record.Field(linha, "T4")) otherwise null,
            try Number.From(Record.Field(linha, "T5")) otherwise null,
            try Number.From(Record.Field(linha, "T6")) otherwise null,
            try Number.From(Record.Field(linha, "T7")) otherwise null,
            try Number.From(Record.Field(linha, "T8")) otherwise null,
            try Number.From(Record.Field(linha, "T9")) otherwise null,
            try Number.From(Record.Field(linha, "T10")) otherwise null
        },

    // Função: blocos consecutivos válidos, mínimo de 2 tiros
    SequenciasValidas = (lista as list, criterio as function) as list =>
        let
            flags =
                List.Transform(
                    lista,
                    each if _ <> null and criterio(_) then 1 else 0
                ),

            acumulado =
                List.Accumulate(
                    flags,
                    [atual = 0, blocos = {}],
                    (estado, valor) =>
                        if valor = 1 then
                            [
                                atual = estado[atual] + 1,
                                blocos = estado[blocos]
                            ]
                        else
                            [
                                atual = 0,
                                blocos =
                                    if estado[atual] >= 2 then
                                        estado[blocos] & {estado[atual]}
                                    else
                                        estado[blocos]
                            ]
                ),

            blocosFinais =
                if acumulado[atual] >= 2 then
                    acumulado[blocos] & {acumulado[atual]}
                else
                    acumulado[blocos]
        in
            blocosFinais,

    QtdSequencias = (lista as list, criterio as function) as number =>
        List.Count(SequenciasValidas(lista, criterio)),

    MaxSequencia = (lista as list, criterio as function) as number =>
        let
            blocos = SequenciasValidas(lista, criterio)
        in
            if List.IsEmpty(blocos) then 0 else List.Max(blocos),

    // TOTAL_SERIE recalculado, sem usar Coluna1 da INPUT
    Add_TOTAL_SERIE =
        Table.AddColumn(
            Tipos,
            "TOTAL_SERIE",
            each List.Sum(List.RemoveNulls(ListaTiros(_))),
            type number
        ),

    // CHAVE_SERIE oficial
    Add_CHAVE_SERIE =
        Table.AddColumn(
            Add_TOTAL_SERIE,
            "CHAVE_SERIE",
            each
                Text.From([DATA_COLETA])
                & "_"
                & Text.From([PROVA])
                & "_"
                & Text.From([ATLETA])
                & "_"
                & Text.From([EVENTO])
                & "_"
                & Text.From([SESSAO])
                & "_"
                & Text.From([ID_BLOCO])
                & "_"
                & Text.From([SERIE]),
            type text
        ),

    // HCI 9/10: tiros >= 9
    Add_HCI_9_10_QTD =
        Table.AddColumn(
            Add_CHAVE_SERIE,
            "HCI_SEQ_9_10_QTD",
            each QtdSequencias(ListaTiros(_), (x) => x >= 9),
            Int64.Type
        ),

    Add_HCI_9_10_MAX =
        Table.AddColumn(
            Add_HCI_9_10_QTD,
            "HCI_SEQ_9_10_MAX",
            each MaxSequencia(ListaTiros(_), (x) => x >= 9),
            Int64.Type
        ),

    // HCI 9: tiros >= 9 e < 10
    Add_HCI_9_QTD =
        Table.AddColumn(
            Add_HCI_9_10_MAX,
            "HCI_SEQ_9_QTD",
            each QtdSequencias(ListaTiros(_), (x) => x >= 9 and x < 10),
            Int64.Type
        ),

    Add_HCI_9_MAX =
        Table.AddColumn(
            Add_HCI_9_QTD,
            "HCI_SEQ_9_MAX",
            each MaxSequencia(ListaTiros(_), (x) => x >= 9 and x < 10),
            Int64.Type
        ),

    // HCI 10: tiros >= 10
    Add_HCI_10_QTD =
        Table.AddColumn(
            Add_HCI_9_MAX,
            "HCI_SEQ_10_QTD",
            each QtdSequencias(ListaTiros(_), (x) => x >= 10),
            Int64.Type
        ),

    Add_HCI_10_MAX =
        Table.AddColumn(
            Add_HCI_10_QTD,
            "HCI_SEQ_10_MAX",
            each MaxSequencia(ListaTiros(_), (x) => x >= 10),
            Int64.Type
        ),

    // Chave interna correta: ATLETA + EVENTO
    // Usada apenas para calcular RESULTADO_EVENTO
    Add_CHAVE_BLOCO_INTERNA =
    Table.AddColumn(
        Add_HCI_10_MAX,
        "CHAVE_BLOCO_INTERNA",
        each
            Text.From([ATLETA])
            & "_"
            & Text.From([PROVA])
            & "_"
            & Text.From([EVENTO])
            & "_"
            & Text.From([SESSAO])
            & "_"
            & Text.From([ID_BLOCO]),
        type text
    ),
	Add_SERIE_ORDER =
    Table.AddColumn(
        Add_CHAVE_BLOCO_INTERNA,
        "HCI_SERIE_ORDER",
        each
            try Number.FromText(
                Text.Select(
                    Text.Upper(Text.Trim(Text.From([SERIE]))),
                    {"0".."9"}
                )
            )
            otherwise null,
        Int64.Type
    ),

FinalPorEvento =
    Table.Group(
        Add_SERIE_ORDER,
        {"CHAVE_BLOCO_INTERNA"},
        {
            {
                "HCI_FINAL_SERIE_ORDER",
                each
                    let
                        finais =
                            Table.SelectRows(
                                _,
                                each Text.Upper(Text.Trim(Text.From([STATUS_EVENTO]))) = "FINAL"
                            ),
                        ordens =
                            List.RemoveNulls(finais[HCI_SERIE_ORDER])
                    in
                        if List.Count(ordens) = 0 then null else List.Min(ordens),
                Int64.Type
            }
        }
    ),

MesclarFinalPorEvento =
    Table.NestedJoin(
        Add_SERIE_ORDER,
        {"CHAVE_BLOCO_INTERNA"},
        FinalPorEvento,
        {"CHAVE_BLOCO_INTERNA"},
        "TB_FINAL_EVENTO",
        JoinKind.LeftOuter
    ),

ExpandirFinalPorEvento =
    Table.ExpandTableColumn(
        MesclarFinalPorEvento,
        "TB_FINAL_EVENTO",
        {"HCI_FINAL_SERIE_ORDER"},
        {"HCI_FINAL_SERIE_ORDER"}
    ),

	Add_EVENT_ROW_VALID =
    	Table.AddColumn(
        ExpandirFinalPorEvento,
        "HCI_EVENT_ROW_VALID",
        each
            let
                ordem = try Number.From([HCI_SERIE_ORDER]) otherwise null,
                final = try Number.From([HCI_FINAL_SERIE_ORDER]) otherwise null
            in
                if final = null then
                    true
                else if ordem = null then
                    false
                else
                    ordem <= final,
        type logical
    ),

    // Soma somente SR1 a SR6 dentro do mesmo ATLETA + EVENTO

    ResultadoPorEvento =
    Table.Group(
        Add_EVENT_ROW_VALID,

            {"CHAVE_BLOCO_INTERNA"},
            {
                {
                    "TOTAL_EVENTO",
                    	each
                        let
                            linhasValidas =
    						Table.SelectRows(
   	     _,
     			each [HCI_EVENT_ROW_VALID] = true
    )

                        in
                            List.Sum(List.RemoveNulls(linhasValidas[TOTAL_SERIE])),
                    type number
                },
                {
                    "TEM_FINAL",
                    each
                        List.Contains(
                            List.Transform(
                                [STATUS_EVENTO],
                                each Text.Upper(Text.Trim(Text.From(_)))
                            ),
                            "FINAL"
                        ),
                    type logical
                }
            }
        ),


    // Junta o total do evento de volta na tabela principal

    MesclarResultadoEvento =
    	Table.NestedJoin(
        Add_EVENT_ROW_VALID,

            {"CHAVE_BLOCO_INTERNA"},
            ResultadoPorEvento,
            {"CHAVE_BLOCO_INTERNA"},
            "TB_RESULTADO_EVENTO",
            JoinKind.LeftOuter
        ),

    ExpandirResultadoEvento =
        Table.ExpandTableColumn(
            MesclarResultadoEvento,
            "TB_RESULTADO_EVENTO",
            {"TOTAL_EVENTO", "TEM_FINAL"},
            {"TOTAL_EVENTO", "TEM_FINAL"}
        ),

    // RESULTADO_EVENTO só aparece na linha FINAL
    Add_RESULTADO_EVENTO =
        Table.AddColumn(
            ExpandirResultadoEvento,
            "RESULTADO_EVENTO",
            each
                let
                    status = Text.Upper(Text.Trim(Text.From([STATUS_EVENTO]))),
                    ehFinal = status = "FINAL",
                    temFinal = try [TEM_FINAL] otherwise false
                in
                    if ehFinal and temFinal then
                        [TOTAL_EVENTO]
                    else
                        null,
            type number
        ),

    // Tabela apenas com resultados finais válidos por atleta
    ResultadosValidos =
        Table.SelectRows(
            Add_RESULTADO_EVENTO,
            each [RESULTADO_EVENTO] <> null
        ),

    // Mediana dos resultados válidos por atleta
    MedianaResultadoPorAtleta =
        Table.Group(
            ResultadosValidos,
            {"ATLETA"},
            {
                {
                    "MED_RESULTADO_ATLETA",
                    each List.Median(List.RemoveNulls([RESULTADO_EVENTO])),
                    type number
                }
            }
        ),

    MesclarMedResultado =
        Table.NestedJoin(
            Add_RESULTADO_EVENTO,
            {"ATLETA"},
            MedianaResultadoPorAtleta,
            {"ATLETA"},
            "TB_MED_RESULTADO",
            JoinKind.LeftOuter
        ),

    ExpandirMedResultado =
        Table.ExpandTableColumn(
            MesclarMedResultado,
            "TB_MED_RESULTADO",
            {"MED_RESULTADO_ATLETA"},
            {"MED_RESULTADO_ATLETA"}
        ),

    // MED_RESULTADO só aparece na linha FINAL válida
    Add_MED_RESULTADO =
        Table.AddColumn(
            ExpandirMedResultado,
            "MED_RESULTADO",
            each
                if [RESULTADO_EVENTO] <> null then
                    [MED_RESULTADO_ATLETA]
                else
                    null,
            type number
        ),

    // Medianas HCI por atleta, considerando somente linhas FINAL válidas
    HCIValidosFinal =
        Table.SelectRows(
            Add_MED_RESULTADO,
            each [RESULTADO_EVENTO] <> null
        ),

    MedianasHCIporAtleta =
        Table.Group(
            HCIValidosFinal,
            {"ATLETA"},
            {
                {"MED_HCI_SEQ_9_10_QTD_ATLETA", each List.Median(List.RemoveNulls([HCI_SEQ_9_10_QTD])), type number},
                {"MED_HCI_SEQ_9_10_MAX_ATLETA", each List.Median(List.RemoveNulls([HCI_SEQ_9_10_MAX])), type number},
                {"MED_HCI_SEQ_9_QTD_ATLETA", each List.Median(List.RemoveNulls([HCI_SEQ_9_QTD])), type number},
                {"MED_HCI_SEQ_9_MAX_ATLETA", each List.Median(List.RemoveNulls([HCI_SEQ_9_MAX])), type number},
                {"MED_HCI_SEQ_10_QTD_ATLETA", each List.Median(List.RemoveNulls([HCI_SEQ_10_QTD])), type number},
                {"MED_HCI_SEQ_10_MAX_ATLETA", each List.Median(List.RemoveNulls([HCI_SEQ_10_MAX])), type number}
            }
        ),

    MesclarMedianasHCI =
        Table.NestedJoin(
            Add_MED_RESULTADO,
            {"ATLETA"},
            MedianasHCIporAtleta,
            {"ATLETA"},
            "TB_MED_HCI",
            JoinKind.LeftOuter
        ),

    ExpandirMedianasHCI =
        Table.ExpandTableColumn(
            MesclarMedianasHCI,
            "TB_MED_HCI",
            {
                "MED_HCI_SEQ_9_10_QTD_ATLETA",
                "MED_HCI_SEQ_9_10_MAX_ATLETA",
                "MED_HCI_SEQ_9_QTD_ATLETA",
                "MED_HCI_SEQ_9_MAX_ATLETA",
                "MED_HCI_SEQ_10_QTD_ATLETA",
                "MED_HCI_SEQ_10_MAX_ATLETA"
            },
            {
                "MED_HCI_SEQ_9_10_QTD_ATLETA",
                "MED_HCI_SEQ_9_10_MAX_ATLETA",
                "MED_HCI_SEQ_9_QTD_ATLETA",
                "MED_HCI_SEQ_9_MAX_ATLETA",
                "MED_HCI_SEQ_10_QTD_ATLETA",
                "MED_HCI_SEQ_10_MAX_ATLETA"
            }
        ),

    // Medianas HCI só aparecem nas linhas FINAL válidas
    Add_MED_HCI_SEQ_9_10_QTD =
        Table.AddColumn(
            ExpandirMedianasHCI,
            "MED_HCI_SEQ_9_10_QTD",
            each if [RESULTADO_EVENTO] <> null then [MED_HCI_SEQ_9_10_QTD_ATLETA] else null,
            type number
        ),

    Add_MED_HCI_SEQ_9_10_MAX =
        Table.AddColumn(
            Add_MED_HCI_SEQ_9_10_QTD,
            "MED_HCI_SEQ_9_10_MAX",
            each if [RESULTADO_EVENTO] <> null then [MED_HCI_SEQ_9_10_MAX_ATLETA] else null,
            type number
        ),

    Add_MED_HCI_SEQ_9_QTD =
        Table.AddColumn(
            Add_MED_HCI_SEQ_9_10_MAX,
            "MED_HCI_SEQ_9_QTD",
            each if [RESULTADO_EVENTO] <> null then [MED_HCI_SEQ_9_QTD_ATLETA] else null,
            type number
        ),

    Add_MED_HCI_SEQ_9_MAX =
        Table.AddColumn(
            Add_MED_HCI_SEQ_9_QTD,
            "MED_HCI_SEQ_9_MAX",
            each if [RESULTADO_EVENTO] <> null then [MED_HCI_SEQ_9_MAX_ATLETA] else null,
            type number
        ),

    Add_MED_HCI_SEQ_10_QTD =
        Table.AddColumn(
            Add_MED_HCI_SEQ_9_MAX,
            "MED_HCI_SEQ_10_QTD",
            each if [RESULTADO_EVENTO] <> null then [MED_HCI_SEQ_10_QTD_ATLETA] else null,
            type number
        ),

    Add_MED_HCI_SEQ_10_MAX =
        Table.AddColumn(
            Add_MED_HCI_SEQ_10_QTD,
            "MED_HCI_SEQ_10_MAX",
            each if [RESULTADO_EVENTO] <> null then [MED_HCI_SEQ_10_MAX_ATLETA] else null,
            type number
        ),

    // =========================================================
    // BLOCO NOVO — HCI MASTER ENGINE
    // Mantém a TB_HCI atual e apenas acrescenta novas colunas
    // =========================================================

    // Garante OBSERVACAO
    Add_OBSERVACAO =
        if Table.HasColumns(Add_MED_HCI_SEQ_10_MAX, "OBSERVACAO") then
            Add_MED_HCI_SEQ_10_MAX
        else
            Table.AddColumn(Add_MED_HCI_SEQ_10_MAX, "OBSERVACAO", each null, type text),

    // ---------------------------------------------------------
    // Funções auxiliares novas
    // ---------------------------------------------------------

    TextoSeguro = (valor as any) as text =>
        if valor = null then "" else Text.Upper(Text.Trim(Text.From(valor))),

    NumeroSeguro = (valor as any) as nullable number =>
        try Number.From(valor) otherwise null,

    StdPopulacional = (lista as list) as nullable number =>
        let
            vals = List.RemoveNulls(List.Transform(lista, each try Number.From(_) otherwise null)),
            n = List.Count(vals),
            media = if n = 0 then null else List.Average(vals),
            variancia =
                if n = 0 then
                    null
                else
                    List.Sum(List.Transform(vals, each Number.Power(_ - media, 2))) / n
        in
            if variancia = null then null else Number.Sqrt(variancia),

    MedianaSegura = (lista as list) as nullable number =>
        let
            vals = List.RemoveNulls(List.Transform(lista, each try Number.From(_) otherwise null))
        in
            if List.Count(vals) = 0 then null else List.Median(vals),

    MediaSegura = (lista as list) as nullable number =>
        let
            vals = List.RemoveNulls(List.Transform(lista, each try Number.From(_) otherwise null))
        in
            if List.Count(vals) = 0 then null else List.Average(vals),

    MaxSeguro = (lista as list) as nullable number =>
        let
            vals = List.RemoveNulls(List.Transform(lista, each try Number.From(_) otherwise null))
        in
            if List.Count(vals) = 0 then null else List.Max(vals),

    MinSeguro = (lista as list) as nullable number =>
        let
            vals = List.RemoveNulls(List.Transform(lista, each try Number.From(_) otherwise null))
        in
            if List.Count(vals) = 0 then null else List.Min(vals),

    Limitar010 = (score as nullable number) as nullable number =>
        if score = null then null
        else if score < 0 then 0
        else if score > 10 then 10
        else Number.Round(score, 2),

    Limitar310 = (score as nullable number) as nullable number =>
        if score = null then null
        else if score < 3 then 3
        else if score > 10 then 10
        else Number.Round(score, 2),

    // Identifica tipo do evento de forma tolerante, usando campos existentes
    Add_EVENT_KIND =
        Table.AddColumn(
            Add_OBSERVACAO,
            "HCI_EVENT_KIND",
            each
                let
                    texto =
                        TextoSeguro([STATUS_EVENTO]) & " " &
                        TextoSeguro([EVENTO]) & " " &
                        TextoSeguro([SESSAO])
                in
                    if Text.Contains(texto, "COMPET") then
                        "COMPETICAO"
                    else if Text.Contains(texto, "SIMUL") then
                        "SIMULADO"
                    else if Text.Contains(texto, "TREINO") or Text.Contains(texto, "TRAIN") then
                        "TREINO"
                    else
                        "NAO_CLASSIFICADO",
            type text
        ),

    Add_IS_RIFLE =
        Table.AddColumn(
            Add_EVENT_KIND,
            "HCI_IS_RIFLE",
            each
                let
                    prova = TextoSeguro([PROVA])
                in
                    Text.Contains(prova, "RIFLE") or Text.Contains(prova, "CARABINA"),
            type logical
        ),

    Add_IS_PISTOL =
        Table.AddColumn(
            Add_IS_RIFLE,
            "HCI_IS_PISTOL",
            each
                let
                    prova = TextoSeguro([PROVA])
                in
                    Text.Contains(prova, "PISTOL") or Text.Contains(prova, "PISTOLA"),
            type logical
        ),

    // ---------------------------------------------------------
    // Chave atleta + prova para medianas corretas por modalidade
    // ---------------------------------------------------------

    Add_CHAVE_ATLETA_PROVA =
        Table.AddColumn(
            Add_IS_PISTOL,
            "HCI_CHAVE_ATLETA_PROVA",
            each Text.From([ATLETA]) & "_" & Text.From([PROVA]),
            type text
        ),

    // ---------------------------------------------------------
    // Resultados finais válidos para medianas por tipo de evento
    // ---------------------------------------------------------

    ResultadosFinaisHCI =
        Table.SelectRows(
            Add_CHAVE_ATLETA_PROVA,
            each [RESULTADO_EVENTO] <> null
        ),

    MedianasResultadoPorAtletaProva =
        Table.Group(
            ResultadosFinaisHCI,
            {"HCI_CHAVE_ATLETA_PROVA"},
            {
                {
                    "HCI_MED_COMPETICOES_ATLETA_PROVA",
                    each
                        MedianaSegura(
                            Table.SelectRows(_, each [HCI_EVENT_KIND] = "COMPETICAO")[RESULTADO_EVENTO]
                        ),
                    type number
                },
                {
                    "HCI_MED_SIMULADOS_ATLETA_PROVA",
                    each
                        MedianaSegura(
                            Table.SelectRows(_, each [HCI_EVENT_KIND] = "SIMULADO")[RESULTADO_EVENTO]
                        ),
                    type number
                },
                {
                    "HCI_MED_TREINOS_ATLETA_PROVA",
                    each
                        MedianaSegura(
                            Table.SelectRows(_, each [HCI_EVENT_KIND] = "TREINO")[RESULTADO_EVENTO]
                        ),
                    type number
                },
                {
                    "HCI_MED_SIM_TREINO_ATLETA_PROVA",
                    each
                        MedianaSegura(
                            Table.SelectRows(
                                _,
                                each [HCI_EVENT_KIND] = "SIMULADO" or [HCI_EVENT_KIND] = "TREINO"
                            )[RESULTADO_EVENTO]
                        ),
                    type number
                }
            }
        ),

    MesclarMedianasHCI_Master =
        Table.NestedJoin(
            Add_CHAVE_ATLETA_PROVA,
            {"HCI_CHAVE_ATLETA_PROVA"},
            MedianasResultadoPorAtletaProva,
            {"HCI_CHAVE_ATLETA_PROVA"},
            "TB_HCI_MASTER_MEDIANAS",
            JoinKind.LeftOuter
        ),

    ExpandirMedianasHCI_Master =
        Table.ExpandTableColumn(
            MesclarMedianasHCI_Master,
            "TB_HCI_MASTER_MEDIANAS",
            {
                "HCI_MED_COMPETICOES_ATLETA_PROVA",
                "HCI_MED_SIMULADOS_ATLETA_PROVA",
                "HCI_MED_TREINOS_ATLETA_PROVA",
                "HCI_MED_SIM_TREINO_ATLETA_PROVA"
            },
            {
                "HCI_MED_COMPETICOES_ATLETA_PROVA",
                "HCI_MED_SIMULADOS_ATLETA_PROVA",
                "HCI_MED_TREINOS_ATLETA_PROVA",
                "HCI_MED_SIM_TREINO_ATLETA_PROVA"
            }
        ),

    Add_HCI_MED_TIRO_SIM_TREINO =
        Table.AddColumn(
            ExpandirMedianasHCI_Master,
            "HCI_MED_TIRO_SIM_TREINO_ATLETA_PROVA",
            each
                if [HCI_MED_SIM_TREINO_ATLETA_PROVA] <> null then
                    Number.Round([HCI_MED_SIM_TREINO_ATLETA_PROVA] / 60, 3)
                else
                    null,
            type number
        ),

    // ---------------------------------------------------------
    // Agrupamento por evento para métricas de prova inteira
    // ---------------------------------------------------------

    EventoHCI =
        Table.Group(
            Add_HCI_MED_TIRO_SIM_TREINO,
            {"CHAVE_BLOCO_INTERNA"},
            {
                {
                    "HCI_EVENT_RECORD",
                    each
                        let
                            LinhasSR =
   				Table.Sort(
      				Table.SelectRows(
            _,
            			each [HCI_EVENT_ROW_VALID] = true
        ),
        			{{"HCI_SERIE_ORDER", Order.Ascending}}
    ),

                            SeriesTotals = List.RemoveNulls(LinhasSR[TOTAL_SERIE]),
                            TirosEvento =
                                List.Combine(
                                    List.Transform(
                                        Table.ToRecords(LinhasSR),
                                        each ListaTiros(_)
                                    )
                                ),

                            TirosValidos = List.RemoveNulls(TirosEvento),

                            SR1 = try SeriesTotals{0} otherwise null,
                            SR2 = try SeriesTotals{1} otherwise null,
                            SR3 = try SeriesTotals{2} otherwise null,
                            SR4 = try SeriesTotals{3} otherwise null,
                            SR5 = try SeriesTotals{4} otherwise null,
                            SR6 = try SeriesTotals{5} otherwise null,

                            SeriesMedian = MedianaSegura(SeriesTotals),
                            SeriesMin = MinSeguro(SeriesTotals),
                            SeriesMax = MaxSeguro(SeriesTotals),
                            SeriesAmplitude =
                                if SeriesMin <> null and SeriesMax <> null then
                                    SeriesMax - SeriesMin
                                else
                                    null,

                            FinalSeriesAvg = MediaSegura({SR5, SR6}),
                            WorstSeries = SeriesMin,

                            ResilienceDropDepth =
                                if SeriesMedian <> null and WorstSeries <> null then

                            EventSeq_9_10_QTD = QtdSequencias(TirosValidos, (x) => x >= 9),
                            EventSeq_9_10_MAX = MaxSequencia(TirosValidos, (x) => x >= 9),
                            EventSeq_10_QTD = QtdSequencias(TirosValidos, (x) => x >= 10),
                            EventSeq_10_MAX = MaxSequencia(TirosValidos, (x) => x >= 10),

                                                atual = try MediasBloco{_} otherwise null

                                                    anterior - atual
                                                else
                                                    0
                                ),

                            StdProva = MedianaSegura(StdsBloco),

                            DropCount =
                                List.Count(
                                    List.Select(
                                        List.Numbers(0, List.Count(Drops)),
                                        each
                                            let
                                                drop = try Drops{_} otherwise null,
                                                std = try StdsBloco{_} otherwise null
                                            in
                                                drop <> null and std <> null and drop >= std and drop > 0
                                    )
                                ),

                            MaxDropDepth = MaxSeguro(Drops),

                                else
                                    null
                        in

                                HCI_SR4_TOTAL = SR4,

                                HCI_SERIES_MEDIAN = SeriesMedian,

    Table.Group(
        Add_EMOTIONAL_BASE_ATLETA,
        {"CHAVE_BLOCO_INTERNA"},
        {
            {
                "HCI_EMOTIONAL_SCORE_BASE_EVENTO",
                each
                    let
                        LinhasSR =
   			 Table.Sort(
       			 Table.SelectRows(
            _,
          		  each [HCI_EVENT_ROW_VALID] = true
        ),
       			 {{"HCI_SERIE_ORDER", Order.Ascending}}
    ),

                        BaseEmotional =
                            try
                                List.First(
                                    List.RemoveNulls(
                                        LinhasSR[HCI_EMOTIONAL_BASE_ATLETA]
                                    )
                                )
                            otherwise
                                null,

                        ProvaTexto =
                            try
                                TextoSeguro(
                                    List.First(
                                        List.RemoveNulls(
                                            LinhasSR[PROVA]
                                        )
                                    )
                                )
                            otherwise
                                "",

                        EhRifle =
                            Text.Contains(ProvaTexto, "RIFLE")
                            or Text.Contains(ProvaTexto, "CARABINA"),

                        ClassificarTiro = (tiro as nullable number) as text =>
                            if tiro = null or BaseEmotional = null then
                                "NULL"
                            else if EhRifle then
                                if tiro >= BaseEmotional then
                                    "LIMPO"
                                else if tiro >= BaseEmotional - 0.2 then
                                    "ALERTA"
                                else
                                    "ERRO"
                            else
                                if tiro >= BaseEmotional then
                                    "LIMPO"
                                else if tiro >= BaseEmotional - 1 then
                                    "ALERTA"
                                else
                                    "ERRO",

                        TodosTiros =
                            List.Combine(
                                List.Transform(
                                    Table.ToRecords(LinhasSR),
                                    each ListaTiros(_)
                                )
                            ),

                        Classes =
                            List.Transform(
                                TodosTiros,
                                each ClassificarTiro(_)
                            ),

                        ListaClasses =
                            List.Select(
                                Classes,
                                each _ <> "NULL"
                            ),

                        Grupos =
                            List.Accumulate(
                                ListaClasses,
                                [atual = {}, grupos = {}],
                                (estado, classe) =>
                                    if classe = "LIMPO" then
                                        [
                                            atual = {},
                                            grupos =
                                                if List.Count(estado[atual]) > 0 then
                                                    estado[grupos] & {estado[atual]}
                                                else
                                                    estado[grupos]
                                        ]
                                    else
                                        [
                                            atual = estado[atual] & {classe},
                                            grupos = estado[grupos]
                                        ]
                            ),

                        GruposFinais =
                            if List.Count(Grupos[atual]) > 0 then
                                Grupos[grupos] & {Grupos[atual]}
                            else
                                Grupos[grupos],

                        PenalidadeGrupo = (grupo as list) as number =>
                            let
                                n = List.Count(grupo),
                                alertas =
                                    List.Count(
                                        List.Select(
                                            grupo,
                                            each _ = "ALERTA"
                                        )
                                    ),
                                erros =
                                    List.Count(
                                        List.Select(
                                            grupo,
                                            each _ = "ERRO"
                                        )
                                    ),

                                penalidade =
                                    if n < 2 then
                                        0
                                    else if erros >= 2 then
                                        if n >= 4 then 2.0 else 1.5
                                    else if erros = 1 then
                                        if n >= 3 then 1.5 else 1.0
                                    else
                                        if alertas >= 5 then 1.5
                                        else if alertas = 4 then 1.0
                                        else if alertas = 3 then 0.5
                                        else 0
                            in
                                penalidade,

                        PenalidadeTotal =
                            List.Sum(
                                List.Transform(
                                    GruposFinais,
                                    each PenalidadeGrupo(_)
                                )
                            ),

                        Score =
                            if BaseEmotional = null or List.Count(ListaClasses) = 0 then
                                null
                            else
                                Limitar310(10 - PenalidadeTotal)
                    in
                        Score,
                type number
            }
        }

shared HCI_RHYTHM_TIMELINE = let
    // =========================================================
    // FONTES DIRETAS
    // =========================================================

    Fonte =
        Excel.CurrentWorkbook(){[Name="TB_HCI"]}[Content],

    Contexto =
        Excel.CurrentWorkbook(){[Name="TB_HCI_SELECTED_CONTEXT"]}[Content],

    // =========================================================
    // FUNÇÕES AUXILIARES
    // =========================================================

    TextoLimpo = (valor as any) as text =>
        try
            if valor = null then "" else Text.Upper(Text.Trim(Text.From(valor)))
        otherwise "",

    CampoTexto = (linha as nullable record, campo as text) as text =>
        try
            if linha = null then
                ""
            else if not Record.HasFields(linha, campo) then
                ""
            else
                let
                    valor = Record.Field(linha, campo)
                in
                    if valor = null then "" else Text.From(valor)
        otherwise
            "",

    NumeroSeguro = (valor as any) as nullable number =>
        try Number.From(valor) otherwise null,

    StdPopulacional = (lista as list) as nullable number =>
        let
            vals =
                List.RemoveNulls(
                    List.Transform(lista, each try Number.From(_) otherwise null)
                ),

            n =
                List.Count(vals),

            media =
                if n = 0 then null else List.Average(vals),

            variancia =
                if n = 0 then
                    null
                else
                    List.Sum(
                        List.Transform(
                            vals,
                            each Number.Power(_ - media, 2)
                        )
                    ) / n
        in
            if variancia = null then null else Number.Sqrt(variancia),

    // =========================================================
    // CONTEXTO SELECIONADO
    // ENVIRONMENT corresponde a HCI_EVENT_KIND na TB_HCI
    // =========================================================

    ContextoLinha =
        try Contexto{0} otherwise null,

    ContextoAtleta =
        CampoTexto(ContextoLinha, "ATLETA"),

    ContextoEvento =
        CampoTexto(ContextoLinha, "EVENTO"),

    ContextoEventType =
        CampoTexto(ContextoLinha, "ENVIRONMENT"),

    ContextoProva =
        CampoTexto(ContextoLinha, "PROVA"),

    // =========================================================
    // BASE LIMPA
    // =========================================================

    BaseHCI =
        Table.SelectRows(
            Fonte,
            each TextoLimpo(CampoTexto(_, "ATLETA")) <> ""
        ),

    // =========================================================
    // FILTRO DIRETO DE CONTEXTO
    // =========================================================

    FiltrarContexto =
        Table.SelectRows(
            BaseHCI,
            each
                TextoLimpo(CampoTexto(_, "ATLETA")) = TextoLimpo(ContextoAtleta)
                and TextoLimpo(CampoTexto(_, "EVENTO")) = TextoLimpo(ContextoEvento)
                and TextoLimpo(CampoTexto(_, "HCI_EVENT_KIND")) = TextoLimpo(ContextoEventType)
                and (
                    TextoLimpo(ContextoProva) = ""
                    or TextoLimpo(CampoTexto(_, "PROVA")) = TextoLimpo(ContextoProva)
                )
                and List.Contains(
                    {"SR1", "SR2", "SR3", "SR4", "SR5", "SR6"},
                    TextoLimpo(CampoTexto(_, "SERIE"))
                )
        ),

    // =========================================================
    // ORDEM DA SÉRIE
    // =========================================================

    Add_DISPLAY_ORDER =
        Table.AddColumn(
            FiltrarContexto,
            "DISPLAY_ORDER",
            each
                try Number.FromText(Text.Select(CampoTexto(_, "SERIE"), {"0".."9"}))
                otherwise null,
            Int64.Type
        ),

    Ordenar =
        Table.Sort(
            Add_DISPLAY_ORDER,
            {{"DISPLAY_ORDER", Order.Ascending}}
        ),

    // =========================================================
    // MÉDIA DA SÉRIE
    // =========================================================

    Add_MEDIA_SERIE =
        Table.AddColumn(
            Ordenar,
            "MEDIA_SERIE",
            each
                let
                    total = NumeroSeguro(Record.FieldOrDefault(_, "TOTAL_SERIE", null))
                in
                    if total <> null then
                        Number.Round(total / 10, 3)
                    else
                        null,
            type nullable number
        ),

    // =========================================================
    // STD DA SÉRIE
    // =========================================================

    Add_STD_SERIE =
        Table.AddColumn(
            Add_MEDIA_SERIE,
            "STD_SERIE",
            each
                let
                    tiros =
                        {
                            NumeroSeguro(Record.FieldOrDefault(_, "T1", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T2", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T3", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T4", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T5", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T6", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T7", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T8", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T9", null)),
                            NumeroSeguro(Record.FieldOrDefault(_, "T10", null))
                        }
                in
                    StdPopulacional(tiros),
            type nullable number
        ),

    // =========================================================
    // MÉDIA DA SÉRIE ANTERIOR
    // =========================================================

    Add_PREV_MEDIA =
        Table.AddColumn(
            Add_STD_SERIE,
            "PREV_MEDIA_SERIE",
            each
                let
                    ordem = [DISPLAY_ORDER],
                    anteriores =
                        Table.SelectRows(
                            Add_STD_SERIE,
                            each [DISPLAY_ORDER] = ordem - 1
                        )
                in
                    if ordem = null or Table.RowCount(anteriores) = 0 then
                        null
                    else
                        anteriores{0}[MEDIA_SERIE],
            type nullable number
        ),

    // =========================================================
    // QUEBRA DE RITMO
    // =========================================================

    Add_BREAK_COUNT =
        Table.AddColumn(
            Add_PREV_MEDIA,
            "BREAK_COUNT",
            each
                if [PREV_MEDIA_SERIE] = null or [MEDIA_SERIE] = null then
                    0
                else if [MEDIA_SERIE] < [PREV_MEDIA_SERIE] then
                    1
                else
                    0,
            Int64.Type
        ),

    Add_MAIN_DROP_DEPTH =
        Table.AddColumn(
            Add_BREAK_COUNT,

                    prev = try Number.From([PREV_MEDIA_SERIE]) otherwise null,

                        if prev = null or atual = null then
                            null

                            prev - atual
                in
                    if queda = null or queda <= 0 then
                        null
                    else
                        Number.Round(queda, 3),
            type nullable number
        ),

    Add_MAIN_DROP_MARKER =
        Table.AddColumn(

            type nullable number

        List.Select(

            each _ > 0
        ),

    MaiorQueda =
        if List.Count(ListaQuedas) = 0 then

                        and queda = MaiorQueda

    // =========================================================

    BaseHCI =
        Table.SelectRows(
            FonteHCI,
            each TextoLimpo(CampoTexto(_, "ATLETA")) <> ""
        ),

    FiltroAtleta =
        Table.SelectRows(
            BaseHCI,
            each TextoLimpo(CampoTexto(_, "ATLETA")) = AtletaSelecionado
        ),

    FiltroEvento =
        Table.SelectRows(
            FiltroAtleta,
            each
                EventoSelecionado = ""
                or TextoLimpo(CampoTexto(_, "EVENTO")) = EventoSelecionado
        ),

    FiltroAmbiente =
        Table.SelectRows(
            FiltroEvento,
            each
                EnvironmentSelecionado = ""
                or TextoLimpo(CampoTexto(_, "HCI_EVENT_KIND")) = EnvironmentSelecionado
        ),

    FiltroProva =
        Table.SelectRows(
            FiltroAmbiente,
            each
                ProvaSelecionadaContexto = ""
                or TextoLimpo(CampoTexto(_, "PROVA")) = ProvaSelecionadaContexto
        ),

    ContextoEscolhido =
    FiltroProva,

    // =========================================================
    // ESCOLHER LINHA FINAL DO EVENTO
    // STATUS_EVENTO = FINAL
    // =========================================================

    LinhasFinal =
        Table.SelectRows(
            ContextoEscolhido,
            each TextoLimpo(CampoTexto(_, "STATUS_EVENTO")) = "FINAL"
        ),

    LinhaFinal =
        if Table.RowCount(LinhasFinal) > 0 then
            LinhasFinal{0}
        else if Table.RowCount(ContextoEscolhido) > 0 then
            ContextoEscolhido{0}
        else
            null,

    // =========================================================
    // SCORES STRUCTURE DIRETO DA TB_HCI
    // =========================================================

    ProvaSelecionada =
        CampoTexto(LinhaFinal, "PROVA"),

    ResultadoEvento =
        CampoNumero(LinhaFinal, "RESULTADO_EVENTO"),

    OutcomeReference =
        CampoNumero(LinhaFinal, "HCI_OUTCOME_REFERENCE"),

    TransferScoreRaw =
        CampoNumero(LinhaFinal, "HCI_TRANSFER_SCORE_BASE"),

    TransferScore =
        if TransferScoreRaw <> null then
            LimitarScore(TransferScoreRaw)
        else if ResultadoEvento <> null
            and OutcomeReference <> null
            and OutcomeReference <> 0
        then
            LimitarScore((((ResultadoEvento / OutcomeReference) * 10) - 9) * 10)
        else
            null,

    ResilienceScore =
        LimitarResilience(CampoNumero(LinhaFinal, "HCI_RESILIENCE_SCORE_BASE")),

    PressureScore =
        LimitarScore(CampoNumero(LinhaFinal, "HCI_PRESSURE_SCORE_BASE")),

    EmotionalScore =
        LimitarEmotional(CampoNumero(LinhaFinal, "HCI_EMOTIONAL_SCORE_BASE")),

    PhysicalScore =
        LimitarScore(CampoNumero(LinhaFinal, "HCI_PHYSICAL_SCORE_BASE")),

    // =========================================================
    // SAÍDA PARA RADIAL STRUCTURE
    // =========================================================

    Resultado =
    #table(
        type table
        [
            DISPLAY_ORDER = Int64.Type,
            PARAMETER = text,
            SCORE = nullable number,
            CURRENT_LEVEL = nullable text,
            REPORT_PROFILE = text
        ],
        {
            {1, "TRANSFER", TransferScore, NivelTransfer(TransferScore, ProvaSelecionada), "STRUCTURE"},
            {2, "RESILIENCE", ResilienceScore, Nivel(ResilienceScore), "STRUCTURE"},
            {3, "PRESSURE", PressureScore, Nivel(PressureScore), "STRUCTURE"},
            {4, "EMOTIONAL", EmotionalScore, NivelEmotional(EmotionalScore), "STRUCTURE"},
            {5, "PHYSICAL", PhysicalScore, Nivel(PhysicalScore), "STRUCTURE"}
        }
    )
in
    Resultado;

TextoLimpo = (valor as any) as text =>
    try
        if valor = null then "" else Text.Upper(Text.Trim(Text.From(valor)))
    otherwise "",

CampoNumero = (linha as nullable record, campo as text) as nullable number =>
    try
        if linha = null then
            null
        else if Record.HasFields(linha, campo) then
            Number.From(Record.Field(linha, campo))
        else
            null
    otherwise
        null,

CampoTexto = (linha as nullable record, campo as text) as text =>
    try
        if linha = null then
            ""
        else if Record.HasFields(linha, campo) then
            Text.From(Record.Field(linha, campo))
        else
            ""
    otherwise
        "",
    LimitarScore = (score as nullable number) as nullable number =>
    if score = null then
        null
    else if score < 0 then
        0
    else if score > 10 then
        10
    else
        Number.Round(score, 2),
    
    Nivel = (score as nullable number) as nullable text =>
    if score = null then
        null
    else if score >= 8.5 then
        "ELITE"
    else if score >= 6 then
        "ALTO RENDIMENTO"
    else if score >= 3 then
        "INTERMEDIÁRIO"
    else
        "INICIANTE",

    RhythmScoreBase10 = (std as nullable number) as nullable number =>
    if std = null then
        null
    else if std <= 0.42 then
        10
    else if std <= 0.50 then
        LimitarScore(10 - ((std - 0.42) / (0.50 - 0.42)) * 1.5)
    else if std <= 0.90 then
        LimitarScore(8.5 - ((std - 0.50) / (0.90 - 0.50)) * 3.5)
    else
        LimitarScore(5 - ((std - 0.90) * 5)
    ),

    NivelRhythm = (std as nullable number) as nullable text =>
    if std = null then
        null
    else if std <= 0.42 then
        "ELITE"
    else if std <= 0.50 then
        "ALTO RENDIMENTO"
    else if std <= 0.90 then
        "INTERMEDIÁRIO"
    else
        "INICIANTE"
    ,

    else

shared TB_HCI_OUTPUT_DECISION_ENGINE = let
    Fonte_KNOWLEDGE =
        Excel.CurrentWorkbook(){[Name="HCI_KNOWLEDGE_PLAN"]}[Content],

    TextoLimpo = (valor as any) as text =>
        if valor = null then "" else Text.Upper(Text.Trim(Text.From(valor))),

    TextoOriginal = (valor as any) as text =>
        if valor = null then "" else Text.Trim(Text.From(valor)),

    NumeroSeguro = (valor as any) as number =>
        try Number.From(valor) otherwise 999999,

    CampoSeguro = (linha as record, nome as text, optional padrao as any) as any =>
        if Record.HasFields(linha, nome)
        then Record.Field(linha, nome)
        else padrao,

    KnowledgeValido =
        Table.SelectRows(
            Fonte_KNOWLEDGE,
            each
                TextoLimpo(CampoSeguro(_, "PARAMETER_1", "")) <> ""
                and TextoLimpo(CampoSeguro(_, "PARAMETER_2", "")) <> ""
        ),

    AddSourceRank =
        Table.AddColumn(
            KnowledgeValido,
            "SOURCE_RANK",
            each NumeroSeguro(CampoSeguro(_, "RANK", null)),
            type number
        ),

    AddDiagnosisKey =
        Table.AddColumn(
            AddSourceRank,
            "DIAGNOSIS_KEY",
            each
                let
                    p1 = TextoLimpo(CampoSeguro(_, "PARAMETER_1", "")),
                    p2 = TextoLimpo(CampoSeguro(_, "PARAMETER_2", "")),
                    lista = List.Sort({p1, p2})
                in
                    if p1 = "" or p2 = ""
                    then ""
                    else lista{0} & "|" & lista{1},
            type text
        ),

    OrdenadoPorKnowledge =
        Table.Sort(
            AddDiagnosisKey,
            {
                {"SOURCE_RANK", Order.Ascending},
                {"DIAGNOSIS_KEY", Order.Ascending}
            }
        ),

    Agrupado =
        Table.Group(
            OrdenadoPorKnowledge,
            {"DIAGNOSIS_KEY"},
            {
                {
                    "MELHOR_DIAGNOSTICO",
                    each Table.FirstN(_, 1),
                    type table
                }
            }
        ),

    Expandido =
        Table.ExpandTableColumn(
            Agrupado,
            "MELHOR_DIAGNOSTICO",
            List.RemoveItems(
                Table.ColumnNames(OrdenadoPorKnowledge),
                {"DIAGNOSIS_KEY"}
            )
        ),

    ReordenarPorPrioridade =
        Table.Sort(
            Expandido,
            {
                {"SOURCE_RANK", Order.Ascending},
                {"DIAGNOSIS_KEY", Order.Ascending}
            }
        ),

    AddOutputRank =
        Table.AddIndexColumn(
            ReordenarPorPrioridade,
            "OUTPUT_RANK",
            1,
            1,
            Int64.Type
        ),

    AddSourceEngine =
        Table.AddColumn(
            AddOutputRank,
            "SOURCE_ENGINE",
            each "HCI_KNOWLEDGE_PLAN",
            type text
        ),

    AddDisplayStatus =
        Table.AddColumn(
            AddSourceEngine,
            "DISPLAY_STATUS",
            each "SHOW",
            type text
        ),

    AddOutputSlot =
        Table.AddColumn(
            AddDisplayStatus,
            "OUTPUT_SLOT",
            each
                if [OUTPUT_RANK] = 1 then "PRIMARY"
                else if [OUTPUT_RANK] = 2 then "SECONDARY"
                else if [OUTPUT_RANK] = 3 then "TERcIARY"
                else "SUPPORT",
            type text
        ),

    AddOutputTitle =
        Table.AddColumn(
            AddOutputSlot,
            "OUTPUT_TITLE",
            each
                let
                    p1 = TextoOriginal(CampoSeguro(_, "PARAMETER_1", "")),
                    l1 = TextoOriginal(CampoSeguro(_, "LEVEL_1", "")),
                    p2 = TextoOriginal(CampoSeguro(_, "PARAMETER_2", "")),
                    l2 = TextoOriginal(CampoSeguro(_, "LEVEL_2", ""))

        Table.SelectColumns(
            AddOutputAction,

                Table.ColumnNames(AddOutputAction),
                {
                    "OUTPUT_RANK",
                    "OUTPUT_SLOT",
                    "DIAGNOSIS_KEY",
                    "PARAMETER_1",
                    "LEVEL_1",
                    "PARAMETER_2",
                    "LEVEL_2",
                    "OUTPUT_TITLE",

shared TB_HCI_ATHLETE_PROFILE_ENGINE = let

    Fonte_PHASE =
        Excel.CurrentWorkbook(){[Name="HCI_SELECTED_PERIODIZATION_PHASE"]}[Content],

    TextoLimpo = (valor as any) as text =>
        if valor = null then "" else Text.Upper(Text.Trim(Text.From(valor))),

    NumeroSeguro = (valor as any) as nullable number =>
        try Number.From(valor) otherwise null,

    CampoSeguro = (linha as record, nome as text, optional padrao as any) as any =>
        if Record.HasFields(linha, nome)
        then Record.Field(linha, nome)
        else padrao,

    FaseSelecionada =
        try Text.Upper(Text.Trim(Text.From(Fonte_PHASE{0}{0}))) otherwise "GENERAL_PREPARATION",

    LoadPercent =
        if FaseSelecionada = "GENERAL_PREPARATION" then 0.70
        else if FaseSelecionada = "SPECIFIC_PREPARATION" then 0.70
        else if FaseSelecionada = "PRE_COMPETITION" then 0.50
        else if FaseSelecionada = "COMPETITION" then 0.50
        else null,

    AddGenreNormalized =
        Table.AddColumn(
            Fonte_PROFILE,
            "GENRE_NORMALIZED",
            each
                let
                    genero = TextoLimpo(CampoSeguro(_, "GENRE", ""))
                in
                    if genero = "F" or genero = "FEMALE" or genero = "FEMININO" then "FEMALE"
                    else if genero = "M" or genero = "MALE" or genero = "MASCULINO" then "MALE"
                    else "",
            type text
        ),

    AddAgeEngine =
        Table.AddColumn(
            AddGenreNormalized,
            "AGE_ENGINE",
            each
                let
                    birth = CampoSeguro(_, "BIRTH DATE", null),
                    ageFromTable = NumeroSeguro(CampoSeguro(_, "AGE", null)),
                    ageCalculated =
                        try
                            Number.RoundDown(
                                Duration.Days(Date.From(DateTime.LocalNow()) - Date.From(birth)) / 365.25
                            )
                        otherwise null
                in
                    if ageFromTable <> null then ageFromTable else ageCalculated,
            type number
        ),

    AddFcMaxEngine =
        Table.AddColumn(
            AddAgeEngine,
            "FC_MAX_ENGINE",
            each
                let
                    age = NumeroSeguro(CampoSeguro(_, "AGE_ENGINE", null)),
                    genero = TextoLimpo(CampoSeguro(_, "GENRE_NORMALIZED", ""))
                in
                    if age = null then null
                    else if genero = "FEMALE" then 206 - (0.88 * age)
                    else if genero = "MALE" then 208 - (0.7 * age)
                    else null,
            type number
        ),

    AddLoadPercent =
        Table.AddColumn(
            AddFcMaxEngine,
            "PHYSICAL_LOAD_PERCENT",
            each LoadPercent,
            type number
        ),

    AddTargetFc =
        Table.AddColumn(
            AddLoadPercent,
            "TARGET_FC",
            each
                let
                    fc = NumeroSeguro(CampoSeguro(_, "FC_MAX_ENGINE", null)),
                    pct = NumeroSeguro(CampoSeguro(_, "PHYSICAL_LOAD_PERCENT", null))
                in
                    if fc <> null and pct <> null then fc * pct else null,
            type number
        ),

    AddBMI =
        Table.AddColumn(
            AddTargetFc,
            "BMI",
            each
                let
                    peso = NumeroSeguro(CampoSeguro(_, "WHEIGHT", null)),
                    altura = NumeroSeguro(CampoSeguro(_, "HEIGHT", null))
                in
                    if peso <> null and altura <> null and altura > 0
                    then peso / Number.Power(altura / 100, 2)

shared TB_HCI_TRAINING_ENGINE = let
    FonteHCI =
        Excel.CurrentWorkbook(){[Name="TB_HCI"]}[Content],

    FonteContext =
        Excel.CurrentWorkbook(){[Name="TB_HCI_SELECTED_CONTEXT"]}[Content],

    FonteStructure =
        Excel.CurrentWorkbook(){[Name="HCI_PERF_STRUCTURE"]}[Content],

    FontePhysicalBI =
        Excel.CurrentWorkbook(){[Name="TB_HCI_PHYSICAL_TRAINING_BI"]}[Content],

    FontePhase =
        try Excel.CurrentWorkbook(){[Name="HCI_SELECTED_PERIODIZATION_PHASE"]}[Content] otherwise #table({"VALUE"}, {{"GENERAL_PREPARATION"}}),

    TextoLimpo = (valor as any) as text =>
        if valor = null then "" else Text.Upper(Text.Trim(Text.From(valor))),

    TextoSeguro = (valor as any) as text =>
        if valor = null then "" else Text.Trim(Text.From(valor)),

    NumeroSeguro = (valor as any) as nullable number =>
        try Number.From(valor) otherwise null,

    CampoSeguro = (linha as record, campo as text, optional padrao as any) as any =>
        if Record.HasFields(linha, campo) then
            Record.Field(linha, campo)
        else
            padrao,

    PegarCampoTexto = (linha as nullable record, nomes as list) as text =>
        let
            existentes =
                if linha = null then
                    {}
                else
                    List.Select(nomes, each Record.HasFields(linha, _)),

            valor =
                if List.Count(existentes) = 0 then
                    ""
                else
                    try Text.From(Record.Field(linha, existentes{0})) otherwise ""
        in
            valor,

    StdPopulacional = (lista as list) as nullable number =>
        let
            vals =
                List.RemoveNulls(
                    List.Transform(lista, each try Number.From(_) otherwise null)
                ),
            n = List.Count(vals),
            media =
                if n = 0 then null else List.Average(vals),
            variancia =
                if n = 0 then
                    null
                else
                    List.Sum(
                        List.Transform(
                            vals,
                            each Number.Power(_ - media, 2)
                        )
                    ) / n
        in
            if variancia = null then null else Number.Sqrt(variancia),

    // =========================================================
    // CONTEXTO OFICIAL
    // =========================================================

    ContextoLinha =
        try FonteContext{0} otherwise null,

    AtletaSelecionado =
        PegarCampoTexto(ContextoLinha, {"ATHLETE", "ATLETA"}),

    EventoSelecionado =
        PegarCampoTexto(ContextoLinha, {"EVENT", "EVENTO"}),

    EnvironmentSelecionado =
        PegarCampoTexto(ContextoLinha, {"ENVIRONMENT", "EVENT_TYPE", "AMBIENTE"}),

    ProvaSelecionada =
        PegarCampoTexto(ContextoLinha, {"PROVA", "PROOF", "EVENT_PROOF"}),

    FaseSelecionadaRaw =
        try FontePhase{0}{0} otherwise "GENERAL_PREPARATION",

    FaseSelecionada =
        TextoLimpo(FaseSelecionadaRaw),

    // =========================================================
    // FILTRAR TB_HCI PELO CONTEXTO
    // =========================================================

    HCI_Contexto =
        Table.SelectRows(
            FonteHCI,
            each
                TextoLimpo(CampoSeguro(_, "ATLETA", CampoSeguro(_, "ATHLETE", ""))) = TextoLimpo(AtletaSelecionado)
                and TextoLimpo(CampoSeguro(_, "EVENTO", CampoSeguro(_, "EVENT", ""))) = TextoLimpo(EventoSelecionado)
                and (
                    EnvironmentSelecionado = ""
                    or TextoLimpo(CampoSeguro(_, "HCI_EVENT_KIND", CampoSeguro(_, "ENVIRONMENT", CampoSeguro(_, "EVENT_TYPE", "")))) = TextoLimpo(EnvironmentSelecionado)
                )
                and List.Contains(
                    {"SR1", "SR2", "SR3", "SR4", "SR5", "SR6"},
                    TextoLimpo(CampoSeguro(_, "SERIE", ""))
                )
        ),

    // =========================================================
    // CALCULAR STD GLOBAL DA PROVA
    // =========================================================

    ListaTiros =
        List.Combine(
            List.Transform(
                Table.ToRecords(HCI_Contexto),
                each
                    List.RemoveNulls(
                        {
                            NumeroSeguro(CampoSeguro(_, "T1", null)),
                            NumeroSeguro(CampoSeguro(_, "T2", null)),
                            NumeroSeguro(CampoSeguro(_, "T3", null)),
                            NumeroSeguro(CampoSeguro(_, "T4", null)),
                            NumeroSeguro(CampoSeguro(_, "T5", null)),
                            NumeroSeguro(CampoSeguro(_, "T6", null)),
                            NumeroSeguro(CampoSeguro(_, "T7", null)),
                            NumeroSeguro(CampoSeguro(_, "T8", null)),
                            NumeroSeguro(CampoSeguro(_, "T9", null)),
                            NumeroSeguro(CampoSeguro(_, "T10", null))
                        }
                    )
            )
        ),

    STD_GLOBAL =
        StdPopulacional(ListaTiros),

    STD_TRIGGER =
        if STD_GLOBAL <> null and STD_GLOBAL > 0.6 then true else false,

    // =========================================================
    // PEGAR PHYSICAL DA STRUCTURE
    // Degradação final = PHYSICAL SCORE < 10
    // =========================================================

    LinhaPhysical =
        try
            Table.SelectRows(
                FonteStructure,
                each TextoLimpo(CampoSeguro(_, "PARAMETER", "")) = "PHYSICAL"
            ){0}
        otherwise
            null,

    PhysicalScore =
        if LinhaPhysical = null then
            null
        else
            NumeroSeguro(CampoSeguro(LinhaPhysical, "SCORE", null)),

    PhysicalLevel =
        if LinhaPhysical = null then
            ""
        else
            TextoSeguro(CampoSeguro(LinhaPhysical, "CURRENT_LEVEL", "")),

    FINAL_DEGRADATION_TRIGGER =
        if PhysicalScore <> null and PhysicalScore < 10 then true else false,

    // =========================================================
    // DEFINIR PRESCRIÇÃO HCI
    // =========================================================

    PhysicalTrainingType =
        if STD_TRIGGER and FINAL_DEGRADATION_TRIGGER then
            "NEUROMUSCULAR + CARDIOPULMONARY"
        else if STD_TRIGGER then
            "NEUROMUSCULAR"
        else if FINAL_DEGRADATION_TRIGGER then
            "CARDIOPULMONARY"
        else
            "NO_PRIORITY_PHYSICAL_PRESCRIPTION",

    PackageNeeded =
        if STD_TRIGGER and FINAL_DEGRADATION_TRIGGER then
            "MIXED_PHYSICAL"
        else if STD_TRIGGER then
            "NEUROMUSCULAR"
        else if FINAL_DEGRADATION_TRIGGER then
            "CARDIOPULMONARY"
        else
            "NONE",

    PhysicalStatus =
        if PhysicalTrainingType = "NO_PRIORITY_PHYSICAL_PRESCRIPTION" then
            "NO_PRIORITY"
        else
            "RECOMMENDED",

    // =========================================================
    // FUNÇÕES PARA ESCOLHER TREINO NA BI FÍSICA
    // =========================================================

    PrepararBI =
        Table.AddColumn(
            FontePhysicalBI,
            "_SORT_PRIORITY",
            each
                let
                    p =
                        NumeroSeguro(CampoSeguro(_, "SUBPACKAGE_PRIORITY", null))
                in
                    if p = null then 999 else p,
            Int64.Type
        ),

    FiltrarPacote = (pacote as text) as table =>
        Table.SelectRows(
            PrepararBI,
            each
                TextoLimpo(CampoSeguro(_, "PHYSICAL_PACKAGE", "")) = TextoLimpo(pacote)
                and TextoLimpo(CampoSeguro(_, "PHYSICAL_BI_STATUS", "")) = "READY_FOR_ENGINE"
                and (
                    CampoSeguro(_, "OUTPUT_ELIGIBLE", true) = true
                    or TextoLimpo(CampoSeguro(_, "OUTPUT_ELIGIBLE", "")) = "TRUE"
                    or TextoLimpo(CampoSeguro(_, "OUTPUT_ELIGIBLE", "")) = "VERDADEIRO"
                    or TextoLimpo(CampoSeguro(_, "OUTPUT_ELIGIBLE", "")) = "SIM"
                )
        ),

    MelhorTreino = (pacote as text) as nullable record =>
        let
            candidatos =
                FiltrarPacote(pacote),

            ordenado =
                Table.Sort(
                    candidatos,
                    {
                        {"_SORT_PRIORITY", Order.Ascending},
                        {"TRAINING_NAME_ORIGINAL", Order.Ascending}
                    }
                ),

            registro =
                if Table.RowCount(ordenado) = 0 then
                    null
                else
                    ordenado{0}
        in
            registro,

    TreinoNeuro =
        MelhorTreino("NEUROMUSCULAR"),

    TreinoCardio =
        MelhorTreino("CARDIOPULMONARY"),

    TreinoMixed =
        MelhorTreino("MIXED_PHYSICAL"),

    TreinoSelecionado =
        if PackageNeeded = "MIXED_PHYSICAL" and TreinoMixed <> null then
            TreinoMixed
        else if PackageNeeded = "NEUROMUSCULAR" then
            TreinoNeuro
        else if PackageNeeded = "CARDIOPULMONARY" then
            TreinoCardio
        else
            null,

    NomeTreino =
        if PhysicalTrainingType = "NEUROMUSCULAR + CARDIOPULMONARY" and TreinoMixed = null then
            Text.Combine(
                List.Select(
                    {
                        if TreinoNeuro <> null then TextoSeguro(CampoSeguro(TreinoNeuro, "NORMALIZED_EXERCISE_NAME", CampoSeguro(TreinoNeuro, "TRAINING_NAME_ORIGINAL", ""))) else null,
                        if TreinoCardio <> null then TextoSeguro(CampoSeguro(TreinoCardio, "NORMALIZED_EXERCISE_NAME", CampoSeguro(TreinoCardio, "TRAINING_NAME_ORIGINAL", ""))) else null
                    },
                    each _ <> null and _ <> ""
                ),
                " + "
            )
        else if TreinoSelecionado <> null then
            TextoSeguro(CampoSeguro(TreinoSelecionado, "NORMALIZED_EXERCISE_NAME", CampoSeguro(TreinoSelecionado, "TRAINING_NAME_ORIGINAL", "")))
        else
            "",

    CategoriaTreino =
        if PhysicalTrainingType = "NEUROMUSCULAR + CARDIOPULMONARY" and TreinoMixed = null then
            "COMBINED_SELECTION"
        else if TreinoSelecionado <> null then
            TextoSeguro(CampoSeguro(TreinoSelecionado, "PHYSICAL_CATEGORY", ""))
        else
            "",

    SubpacoteTreino =
        if PhysicalTrainingType = "NEUROMUSCULAR + CARDIOPULMONARY" and TreinoMixed = null then
            Text.Combine(
                List.Select(
                    {
                        if TreinoNeuro <> null then TextoSeguro(CampoSeguro(TreinoNeuro, "PHYSICAL_SUBPACKAGE", "")) else null,
                        if TreinoCardio <> null then TextoSeguro(CampoSeguro(TreinoCardio, "PHYSICAL_SUBPACKAGE", "")) else null
                    },
                    each _ <> null and _ <> ""
                ),
                " + "
            )
        else if TreinoSelecionado <> null then
            TextoSeguro(CampoSeguro(TreinoSelecionado, "PHYSICAL_SUBPACKAGE", ""))
        else
            "",

    // =========================================================
    // CARGA E VOLUME POR FASE
    // =========================================================

    PhaseLoadPercent =
        if List.Contains({"GENERAL_PREPARATION", "SPECIFIC_PREPARATION"}, FaseSelecionada) then
            70
        else if List.Contains({"PRE_COMPETITION", "COMPETITION"}, FaseSelecionada) then
            50
        else
            70,

    LoadRule =
        if PhysicalTrainingType = "NEUROMUSCULAR" then
            Text.From(PhaseLoadPercent) & "% reference load | controlled strength, posture and stability"
        else if PhysicalTrainingType = "CARDIOPULMONARY" then
            Text.From(PhaseLoadPercent) & "% reference volume | aerobic/cardiopulmonary endurance"
        else if PhysicalTrainingType = "NEUROMUSCULAR + CARDIOPULMONARY" then
            Text.From(PhaseLoadPercent) & "% combined reference load | split neuromuscular and cardiopulmonary stimulus"
        else
            "",

    VolumeRule =
        if PhysicalTrainingType = "NEUROMUSCULAR" then
            "2-4 sets | 20-45 seconds isometry or 8-12 slow controlled repetitions"
        else if PhysicalTrainingType = "CARDIOPULMONARY" then
            "20-40 minutes | zone 2 or low-impact aerobic work"
        else if PhysicalTrainingType = "NEUROMUSCULAR + CARDIOPULMONARY" then
            "15-20 minutes neuromuscular + 15-25 minutes cardiopulmonary"
        else
            "",

    SessionDuration =
        if PhysicalTrainingType = "NEUROMUSCULAR" then
            "20-30"
        else if PhysicalTrainingType = "CARDIOPULMONARY" then
            "20-40"
        else if PhysicalTrainingType = "NEUROMUSCULAR + CARDIOPULMONARY" then
            "30-45"
        else
            "",

    Objective =
        if PhysicalTrainingType = "NEUROMUSCULAR" then
            "Reduce global shot variability by improving postural stability, core control and neuromuscular endurance."
        else if PhysicalTrainingType = "CARDIOPULMONARY" then
            "Reduce late-match degradation by improving physiological sustainability and fatigue tolerance."
        else if PhysicalTrainingType = "NEUROMUSCULAR + CARDIOPULMONARY" then
            "Reduce global variability and late-match degradation through combined stability and cardiopulmonary work."
        else
            "No priority physical prescription based on current HCI triggers.",

    PrescriptionText =
        if PhysicalTrainingType = "NO_PRIORITY_PHYSICAL_PRESCRIPTION" then
            "No priority physical training prescription. Maintain regular physical preparation and monitor future HCI changes."
        else
            PhysicalTrainingType
            & " | "
            & Objective
            & " | Suggested exercise: "
            & NomeTreino
            & " | Load: "
            & LoadRule
            & " | Volume: "
            & VolumeRule,

    ProfessionalValidation =
        if PhysicalTrainingType = "NO_PRIORITY_PHYSICAL_PRESCRIPTION" then
            "NOT_REQUIRED_FOR_NO_PRIORITY"
        else
            "RECOMMENDED_PROFESSIONAL_VALIDATION",

    Resultado =
        #table(
            type table
            [
                ATHLETE = text,
                EVENT = text,
                ENVIRONMENT = text,
                PROVA = text,
                PERIODIZATION_PHASE = text,
                STD_GLOBAL = nullable number,
                STD_TRIGGER = logical,
                PHYSICAL_SCORE = nullable number,
                PHYSICAL_LEVEL = text,
                FINAL_DEGRADATION_TRIGGER = logical,
                PHYSICAL_TRAINING_TYPE = text,
                PHYSICAL_PACKAGE_SELECTED = text,
                TRAINING_NAME = text,
                PHYSICAL_CATEGORY = text,
                PHYSICAL_SUBPACKAGE = text,
                PHASE_LOAD_PERCENT = Int64.Type,
                SESSION_DURATION_MIN = text,
                LOAD_RULE = text,
                VOLUME_RULE = text,
                PHYSICAL_TRAINING_OBJECTIVE = text,
                PHYSICAL_TRAINING_PRESCRIPTION = text,
                PHYSICAL_TRAINING_STATUS = text,
                PROFESSIONAL_VALIDATION_STATUS = text
            ],
            {
                {
                    AtletaSelecionado,
                    EventoSelecionado,
                    EnvironmentSelecionado,
                    ProvaSelecionada,
                    FaseSelecionada,
                    if STD_GLOBAL = null then null else Number.Round(STD_GLOBAL, 3),
                    STD_TRIGGER,
                    PhysicalScore,
                    PhysicalLevel,
                    FINAL_DEGRADATION_TRIGGER,
                    PhysicalTrainingType,
                    PackageNeeded,
                    NomeTreino,
                    CategoriaTreino,
                    SubpacoteTreino,
                    PhaseLoadPercent,
                    SessionDuration,
                    LoadRule,
                    VolumeRule,
                    Objective,
                    PrescriptionText,
                    PhysicalStatus,
                    ProfessionalValidation
                }
            }
        )
in
    Resultado;

shared TB_HCI_PHYSICAL_TRAINING_PACKAGE_OUTPUT = let
    FonteEngine =
        Excel.CurrentWorkbook(){[Name="TB_HCI_PHYSICAL_TRAINING_ENGINE"]}[Content],

    FontePackages =
        Excel.CurrentWorkbook(){[Name="TB_HCI_PHYSICAL_TRAINING_PACKAGES"]}[Content],

    TextoLimpo = (valor as any) as text =>
        if valor = null then "" else Text.Upper(Text.Trim(Text.From(valor))),

    TextoSeguro = (valor as any) as text =>
        if valor = null then "" else Text.Trim(Text.From(valor)),

    NumeroSeguro = (valor as any) as nullable number =>
        try Number.From(valor) otherwise null,

    CampoSeguro = (linha as record, campo as text, optional padrao as any) as any =>
        if Record.HasFields(linha, campo) then
            Record.Field(linha, campo)
        else
            padrao,

    // =========================================================
    // LER LINHA ÚNICA DA ENGINE
    // =========================================================

    EngineLinha =
        try FonteEngine{0} otherwise null,

    PhysicalTrainingType =
        if EngineLinha = null then
            ""
        else
            TextoLimpo(CampoSeguro(EngineLinha, "PHYSICAL_TRAINING_TYPE", "")),

    PackageNeeded =
        if PhysicalTrainingType = "NEUROMUSCULAR + CARDIOPULMONARY" then
            "MIXED_PHYSICAL"
        else if PhysicalTrainingType = "NEUROMUSCULAR" then
            "NEUROMUSCULAR"
        else if PhysicalTrainingType = "CARDIOPULMONARY" then
            "CARDIOPULMONARY"
        else
            "NONE",

    PhaseSelected =
        if EngineLinha = null then
            ""
        else
            TextoLimpo(CampoSeguro(EngineLinha, "PERIODIZATION_PHASE", "")),

    // =========================================================
    // PREPARAR PACOTES
    // =========================================================

    PackagesComPrioridade =
        Table.AddColumn(
            FontePackages,
            "_PACKAGE_PRIORITY_NUM",
            each
                let
                    p = NumeroSeguro(CampoSeguro(_, "PACKAGE_PRIORITY", null))
                in
                    if p = null then 999 else p,
            Int64.Type
        ),

    PacotesFiltrados =
        Table.SelectRows(
            PackagesComPrioridade,
            each
                TextoLimpo(CampoSeguro(_, "PHYSICAL_PACKAGE", "")) = PackageNeeded
        ),

    PacotesOrdenados =
        Table.Sort(
            PacotesFiltrados,
            {
                {"_PACKAGE_PRIORITY_NUM", Order.Ascending},
                {"PACKAGE_ID", Order.Ascending}
            }
        ),

    PacoteSelecionado =
        if PackageNeeded = "NONE" or Table.RowCount(PacotesOrdenados) = 0 then
            null
        else
            PacotesOrdenados{0},

    // =========================================================
    // CARGA POR FASE
    // =========================================================

    PhaseLoad =
        if PacoteSelecionado = null then
            ""
        else if PhaseSelected = "GENERAL_PREPARATION" then
            TextoSeguro(CampoSeguro(PacoteSelecionado, "GENERAL_PREPARATION_LOAD", ""))
        else if PhaseSelected = "SPECIFIC_PREPARATION" then
            TextoSeguro(CampoSeguro(PacoteSelecionado, "SPECIFIC_PREPARATION_LOAD", ""))
        else if PhaseSelected = "PRE_COMPETITION" then
            TextoSeguro(CampoSeguro(PacoteSelecionado, "PRE_COMPETITION_LOAD", ""))
        else if PhaseSelected = "COMPETITION" then
            TextoSeguro(CampoSeguro(PacoteSelecionado, "COMPETITION_LOAD", ""))
        else
            "",

    // =========================================================
    // SAÍDA — SEGUNDA LINHA DA OUTPUT
    // =========================================================

    Resultado =
        #table(
            type table
            [
                PACKAGE = text,
                SESSION = text,
                EXERCISES = text,
                EQUIPMENT = text,
                SETS = text,
                REPS_TIME = text,
                REST = text,
                HR_TARGET = text,
                FREQUENCY = text,
                DURATION = text,
                PHASE_LOAD = text,
                OUTPUT_TEXT = text,
                SAFETY_NOTE = text
            ],
            {
                {
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "PACKAGE_ID", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "SESSION_NAME", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "EXERCISES", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "EQUIPMENT", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "SETS", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "REPS_OR_TIME", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "REST_INTERVAL", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "HEART_RATE_TARGET", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "FREQUENCY_WEEK", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "SESSION_DURATION_MIN", "")),
                    PhaseLoad,
                    if PacoteSelecionado = null then
                        "No priority physical package selected."
                    else
                        TextoSeguro(CampoSeguro(PacoteSelecionado, "OUTPUT_TEXT", "")),
                    if PacoteSelecionado = null then "" else TextoSeguro(CampoSeguro(PacoteSelecionado, "SAFETY_NOTE", ""))
                }
            }
        )
in
    Resultado;
